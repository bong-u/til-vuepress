# Controller

@Controller
public class HelloController {
    @GetMapping("/hello")
    public String hello(Model model,
        @RequestParam(value="name", required=false) String name) {
            model.addAttribute("greeting", "안녕하세요" + name);
            return "hello";
    }
}
